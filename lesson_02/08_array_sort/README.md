for(int i = 0; i < N; ++i)

Задание: отсортировать массив
{-2, 10, 1, 5, 6, 11}
{-2, [ 1, 10, 5, 6, 11 ] }
{-2, 1, [5, 10, 6, 11 ] }
SWAP
int temp = A[i];
A[i] = A[i+1];
A[i+1] = temp;
A[i] ^= A[i+1] ^= A[i] ^= A[i+1];
